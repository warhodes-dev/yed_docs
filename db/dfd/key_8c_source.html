285int yed_read_key_sequences(int len, int *input) {
286 int seq_key,
287 i,
288 found_a_partial_match,
289 keep_reading;
290 char c;
291 int new_key;
292 yed_key_sequence *seq_it;
293
294 if (len == 0) { return 0; }
295
296 new_key = (input[len - 1] == CTRL_H && ctrl_h_is_bs
297 ? BACKSPACE
298 : input[len - 1]);
299
300 len -= 1;
301
302 do {
303 /* We have consumed a keystroke. */
304 if (new_key == CTRL_H && ctrl_h_is_bs) {new_key = BACKSPACE; }
305
306 if (new_key == KEY_NULL) {
307 keep_reading = 1;
308 continue;
309 }
310
311 input[len] = new_key;
312 len += 1;
313 keep_reading = 0;
314
315 /*
316 * Should we consume another? See if there's
317 * a key sequence defined that matches the keys
318 * we have so far.
319 */
320 array_traverse(ys->key_sequences, seq_it) {
321 /* Only if there's more in the sequence to read.. */
322 if (seq_it->len > len) {
323 found_a_partial_match = 1;
324 for (i = 0; i < len; i += 1) {
325 if (seq_it->keys[i] != input[i]) {
326 found_a_partial_match = 0;
327 break;
328 }
329 }
330
331 if (found_a_partial_match) {
332 keep_reading = 1;
333 break;
334 }
335 }
336 }
337 } while (keep_reading && read(0, &c, 1) && ((new_key = c), len < MAX_SEQ_LEN));
338
339 seq_key = yed_get_key_sequence(len, input);
340
341 if (seq_key != KEY_NULL) {
342 input[0] = seq_key;
343 return 1;
344 }
345
346 return len;

349int yed_read_mbyte_keys(char first_byte, int n_bytes) {
350 int i;
351 char c;
352
353 ASSERT(n_bytes > 1, "should not read more bytes for single byte char");
354
355 ys->mbyte.bytes[0] = first_byte;
356
357 for (i = 1; i < n_bytes; i += 1) {
358 /*
359 * Check if the read times out.
360 * If this is the case, we're gonna indicate to
361 * the caller that we could not get all of the bytes
362 * that we needed.

367 }

372static int _yed_read_keys(int *input) {
373 int len;
374 int nread;
375 char c;
376 int n_bytes;
377 yed_glyph g;
378
379 len = 0;
380
381 ctrl_h_is_bs = yed_var_is_truthy("ctrl-h-is-backspace");
382
383/*
384 * BLOCKING(ish):
385 **********************************************/
386/* while ((nread = read(0, &c, 1)) == 0); */
387/**********************************************/
388
389/*
390 * NON-BLOCKING:
391 **********************************************/
392 nread = read(0, &c, 1);
393 if (nread == 0) { return 0; }
394/**********************************************/
395
396
397 if (nread == -1) { return 0; }
398
399 n_bytes = nread;
400
401 if (c != 0) {
402 if (c == CTRL_H && ctrl_h_is_bs) { c = BACKSPACE; }
403
404 g.c = c;
405 n_bytes = yed_get_glyph_len(g);
406 }
407
408 if (c == ESC) {
409 input[0] = c;
410
411 len = esc_timeout(input);
412
413 if (len == 3) {
414 len = esc_sequence(input);
415 if (len > 1) { goto do_seq; }
416 }
417 } else if (n_bytes > 1) {
418 if (yed_read_mbyte_keys(g.c, n_bytes)) {
419 input[0] = MBYTE;
420 len = 1;
421 }
422 } else if (c > 0) {
423 input[0] = c;
424 len = 1;
425
426do_seq:;
427 if (!ys->interactive_command && !ys->doing_bracketed_paste) {
428 len = yed_read_key_sequences(len, input);
429 }
430 }
431
432 return len;

505void yed_take_key(int key) {
506 yed_key_binding *binding;
507 char key_str_buff[32];
508 char *key_str;

510

528 key_str = key_str_buff;
529

532 } else if (binding) {

534 } else if (key > 0 && key < REAL_KEY_MAX
535 && (key == ENTER || key == TAB || key == MBYTE || !iscntrl(key))) {

624void yed_bind_key(yed_key_binding binding) {
625 yed_event event;
626 yed_key_binding *b;
627 char **args;
628 int i;
629
630 LOG_FN_ENTER();
631
632 if (binding.key == KEY_NULL) { goto out; }
633
634 memset(&event, 0, sizeof(event));
635 event.kind = EVENT_KEY_PRE_BIND;
636 event.key = binding.key;
637 event.cmd_name = binding.cmd;
638 event.n_args = binding.n_args;
639 event.args = (const char * const*)binding.args;
640 yed_trigger_event(&event);
641 if (event.cancel) { return; }
642
643 if (binding.key == CTRL_H
644 && yed_var_is_truthy("ctrl-h-is-backspace")) {
645 yed_log("[!] warning: Binding ctrl-h to '%s' while 'ctrl-h-is-backspace' is set.\n"
646 " Pressing backspace may invoke this binding depending on what your "
647 "terminal sends when you press backspace.", binding.cmd);
648 }
649
650 yed_unbind_key(binding.key);
651
652 binding.cmd = strdup(binding.cmd);
653 if (binding.n_args) {
654 args = binding.args;
655 binding.args = malloc(sizeof(char*) * binding.n_args);
656 for (i = 0; i < binding.n_args; i += 1) {
657 binding.args[i] = strdup(args[i]);
658 }
659 }
660
661 b = malloc(sizeof(*b));
662 *b = binding;
663
664 if (binding.key < REAL_KEY_MAX) {
665 ys->real_key_map[binding.key] = b;
666 } else {
667 tree_insert(ys->vkey_binding_map, binding.key, b);
668 }
669
670 event.kind = EVENT_KEY_POST_BIND;
671 yed_trigger_event(&event);
672
673out:
674 LOG_EXIT();
675}

677void yed_unbind_key(int key) {
678 yed_event event;
679 tree_it(int, yed_key_binding_ptr_t) it;
680 yed_key_binding *old_binding;
681 int i;
682
683 if (key == KEY_NULL) { return; }
684
685 memset(&event, 0, sizeof(event));
686 event.kind = EVENT_KEY_PRE_UNBIND;
687 event.key = key;
688 yed_trigger_event(&event);
689 if (event.cancel) { return; }
690
691 if (key < REAL_KEY_MAX) {
692 old_binding = ys->real_key_map[key];
693 if (old_binding) {
694 free(old_binding->cmd);
695 if (old_binding->n_args) {
696 for (i = 0; i < old_binding->n_args; i += 1) {
697 free(old_binding->args[i]);
698 }
699 free(old_binding->args);
700 }
701 free(old_binding);
702 ys->real_key_map[key] = NULL;
703 }
704 } else {
705 it = tree_lookup(ys->vkey_binding_map, key);
706
707 if (tree_it_good(it)) {
708 old_binding = tree_it_val(it);
709 tree_delete(ys->vkey_binding_map, tree_it_key(it));
710 free(old_binding->cmd);
711 if (old_binding->n_args) {
712 for (i = 0; i < old_binding->n_args; i += 1) {
713 free(old_binding->args[i]);
714 }
715 free(old_binding->args);
716 }
717 free(old_binding);
718 }
719 }
720
721 event.kind = EVENT_KEY_POST_UNBIND;
722 yed_trigger_event(&event);

938int _yed_string_to_keys(const char *str, int *keys, int allow_meta) {

949 int fn_nr;

951 str_dup = strdup(str);
952 key_str = str_dup;
953 key_ptr = keys;
954 n_keys = 0;
955
956 while (*key_str) {
957 if (n_keys == MAX_SEQ_LEN) {
958 n_keys = -2;
959 goto out;
960 }
961
962 if (is_space(*key_str)) {
963 key_str += 1;
964 continue;
965 }
966
967 scan = strchr(key_str, ' ' );
968 if (!scan) { scan = strchr(key_str, '\t'); }
969 if (!scan) { scan = strchr(key_str, '\n'); }
970 if (!scan) { scan = strchr(key_str, '\r'); }
971
972 if (scan) { save_wspc = *scan; *scan = 0; }
973
974 key_c = key_i = -1;
975 meta = 0;
976 fn_nr = -1;
977
978 if (strlen(key_str) == 1) {
979 sscanf(key_str, "%c", &key_c);
980 key_i = key_c;
981 } else if (strcmp(key_str, "tab") == 0) {
982 key_i = TAB;
983 } else if (strcmp(key_str, "enter") == 0) {
984 key_i = ENTER;
985 } else if (strcmp(key_str, "esc") == 0) {
986 key_i = ESC;
987 } else if (strcmp(key_str, "spc") == 0) {
988 key_i = ' ';
989 } else if (strcmp(key_str, "bsp") == 0) {
990 key_i = BACKSPACE;
991 } else if (strcmp(key_str, "left") == 0) {
992 key_i = ARROW_LEFT;
993 } else if (strcmp(key_str, "right") == 0) {
994 key_i = ARROW_RIGHT;
995 } else if (strcmp(key_str, "up") == 0) {
996 key_i = ARROW_UP;
997 } else if (strcmp(key_str, "down") == 0) {
998 key_i = ARROW_DOWN;
999 } else if (strcmp(key_str, "del") == 0) {
1000 key_i = DEL_KEY;
1001 } else if (strcmp(key_str, "home") == 0) {
1002 key_i = HOME_KEY;
1003 } else if (strcmp(key_str, "end") == 0) {
1004 key_i = END_KEY;
1005 } else if (strcmp(key_str, "pageup") == 0) {
1006 key_i = PAGE_UP;
1007 } else if (strcmp(key_str, "pagedown") == 0) {
1008 key_i = PAGE_DOWN;
1009 } else if (strcmp(key_str, "shift-tab") == 0) {
1010 key_i = SHIFT_TAB;
1011 } else if (strcmp(key_str, "menu") == 0) {
1012 key_i = MENU_KEY;
1013 } else if (sscanf(key_str, "ctrl-%c", &key_c)) {
1014 if (key_c != -1) {
1015 if (key_c == '/') {
1016 key_i = CTRL_FS;
1017 } else {
1018 key_i = CTRL_KEY(key_c);
1019 }
1020 }
1021 } else if (allow_meta && strncmp(key_str, "meta-", 5) == 0) {
1022 meta = 1;
1023
1024 key_i = ESC;
1025 *key_ptr = key_i;
1026 key_ptr += 1;
1027 n_keys += 1;
1028
1029 if (n_keys == MAX_SEQ_LEN) {
1030 n_keys = -2;
1031 goto out;
1032 }
1033
1034 meta_result = _yed_string_to_keys(key_str + 5, key_ptr, 0);
1035
1036 if (meta_result < 0) {
1037 n_keys = meta_result;
1038 goto out;
1039 }
1040
1041 key_ptr += meta_result;
1042 n_keys += meta_result;
1043 } else if (sscanf(key_str, "fn-%d", &fn_nr)) {
1044 if (fn_nr != -1) {
1045 if (fn_nr > 0 && fn_nr < 13) {
1046 key_i = FN1 + (fn_nr - 1);
1047 }
1048 }
1049 }
1050
1051 if (key_i == -1) {
1052 n_keys = -1;
1053 goto out;
1054 }
1055
1056 if (!meta) {
1057 *key_ptr = key_i;
1058 key_ptr += 1;
1059 n_keys += 1;
1060 }
1061
1062 if (scan) {
1063 *scan = save_wspc;
1064 key_str = scan + 1;
1065 } else {
1066 key_str += strlen(key_str);
1067 }
1068 }
1069
1070out:;
1071 free(str_dup);
1072 return n_keys;
1073}
