340int compute_common_prefix_len(char *in, int n_items, char **items) {
341 int i, len, max_len;
342
343 if (n_items == 0) { return 0; }

345 len = strlen(in);
346 max_len = strlen(items[0]);

350 if (items[i][len] != items[0][len]) {
351 goto out;
352 }

354 len += 1;
355 }

361int get_cmd_completion(char *in, char ***out) {
362 tree_it(yed_command_name_t, yed_command) it;
363 int in_len,
364 i,
365 n_items;
366 char **items,
367 *key;
368
369 if (!in) { return 0; }
370
371 in_len = strlen(in);
372 n_items = 0;
373
374 /*
375 * Count the number of commands that start with our 'in'
376 * string. 'in' might be an empty string, in which case,
377 * this method still gives us every command as a completion
378 * item (which is what we want).
379 */
380 it = tree_gtr(ys->commands, in);
381
382 if (!tree_it_good(it)) { goto none; }
383
384 i = 0;
385 key = tree_it_key(it);
386 while (strncmp(key, in, in_len) == 0) {
387 i += 1;
388 tree_it_next(it);
389 if (!tree_it_good(it)) {
390 break;
391 }
392 key = tree_it_key(it);
393 }
394
395 n_items = i;
396
397 if (n_items) {
398 items = malloc(i * sizeof(char*));
399
400 /* Now start collecting the command names. */
401 it = tree_gtr(ys->commands, in);
402 i = 0;
403 while (i < n_items) {
404 items[i] = tree_it_key(it);
405 tree_it_next(it);
406 i += 1;
407 }
408 } else {
409none:
410 items = NULL;
411 }
412
413 *out = items;
414 return n_items;
415}
