7int load_init(const char *path) {
8 int loaded;
9 int err;
10 char full_path[4096];
11 char *msg;
12
13 LOG_FN_ENTER();
14
15 loaded = 0;
16
17 if (!path) { goto not_found; }
18
19 yed_cprint("attempting to load init plugin from '%s'", path);
20
21 snprintf(full_path, sizeof(full_path), "%s/init.so", path);
22 if (access(full_path, F_OK) != -1) {
23 yed_add_plugin_dir(path);
24
25 err = yed_load_plugin("init");
26
27 switch (err) {
28 case YED_PLUG_NOT_FOUND:
29 goto not_found;
30 case YED_PLUG_DLOAD_FAIL:
31 msg = dlerror();
32 if (msg == NULL) {
33 msg = "failed to load init plugin for unknown reason";
34 }
35 break;
36 case YED_PLUG_SUCCESS:
37 msg = "loaded init";
38 break;
39 case YED_PLUG_NO_BOOT:
40 msg = "init missing 'yed_plugin_boot'";
41 break;
42 case YED_PLUG_BOOT_FAIL:
43 msg = "init 'yed_plugin_boot' failed";
44 break;
45 case YED_PLUG_VER_MIS:
46 msg = "init plugin was rejected because it was compiled against an older version of yed and is not compatible with this version";
47 break;
48 }
49
50 if (err == YED_PLUG_SUCCESS) {
51 yed_cprint("\n%s", msg);
52 loaded = 1;
53 } else {
54 yed_cerr("%s", msg);
55 }
56 } else {
57not_found:;
58 yed_cerr("init plugin not found at '%s'", path);
59 }
60
61 LOG_EXIT();
62
63 return loaded;
64}
