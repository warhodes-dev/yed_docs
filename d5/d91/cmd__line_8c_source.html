76 ys->cmd_cursor_x -= 1;

79 case ARROW_RIGHT:

81 ys->cmd_cursor_x += 1;

85 case CTRL_A:
86 ys->cmd_cursor_x = 1 + prompt_width;
87 break;
88 case END_KEY:
89 case CTRL_E:
90 ys->cmd_cursor_x = 1 + prompt_width + width;
91 break;
92 case BACKSPACE:
93 if (width && cursor_idx > 0) {
94 yed_cmd_buff_delete(cursor_idx - 1);
95 }
96 break;
97 case DEL_KEY:

100 ys->cmd_cursor_x += 1;
101 }
102 break;
103
104 /* Explicitly ignored: */
105 case ESC:
106 break;
107
108 default:
109 if (key < ASCII_KEY_MAX && !iscntrl(key)) {
110 yed_cmd_buff_insert(cursor_idx, key);
111 }
112 }
