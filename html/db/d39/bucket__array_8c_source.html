166void * bucket_insert(bucket_array_t *array, int b_idx, int idx, void *elem, int elem_size) {
167 void *elem_slot;
168 bucket_t *b, *spill_b, *next_b, new_b;
169
170 b = array_item(array->buckets, b_idx);
171
172 if (b->used == b->capacity) {
173 next_b = NULL;
174 /*
175 * If there's a next bucket that isn't full, we can spill there.
176 * Otherwise, we need to make a new bucket.
177 */
178 if (b_idx < array_len(array->buckets) - 1) {
179 next_b = array_item(array->buckets, b_idx + 1);
180 }
181
182 if (next_b && next_b->used < next_b->capacity) {
183 spill_b = next_b;
184 } else {
185 /* Make a new empty bucket. */
186 new_b = new_bucket(array);
187 spill_b = array_insert(array->buckets, b_idx + 1, new_b);
188 }
189
190 if (spill_b->used) {
191 /* Shift everything in the spill bucket down an element. */
192 memmove(spill_b->data + elem_size,
193 spill_b->data,
194 elem_size * spill_b->used);
195 }
196
197 spill_b->used += 1;
198
199 /* Move the last element in the current bucket
200 * to be the first element in the next bucket. */
201 memcpy(spill_b->data,
202 b->data + (elem_size * (b->used - 1)),
203 elem_size);
204
205 b->used -= 1;
206 }
207
208 /*
209 * By this point, we (better) have room in the current bucket
210 * to insert an element.
211 */
212 elem_slot = b->data + (elem_size * idx);
213
214 if (idx < b->used) {
215 memmove(elem_slot + elem_size,
216 elem_slot,
217 elem_size * (b->used - idx));
218 }
219
220 memcpy(elem_slot, elem, elem_size);
221
222 b->used += 1;
223 array->used += 1;
224
225 return elem_slot;
226}
