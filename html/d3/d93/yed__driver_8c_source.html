96/*
97 * If some doofus decides to load libyed.so from
98 * a plugin for some crazy reason, they will remove our
99 * ability to do any dynamic reloading because they've
100 * grabbed another reference to the library.
101 * As long as that other reference exists, the system
102 * won't unload our mapped image...
103 *
104 * So, this function will essentially acquire
105 * _all_ of the references to the library one by one
106 * and remove them so that the ref count eventually goes
107 * to zero and the library is unloaded.
108 *
109 * This works, but it will probably screw up whoever took
110 * the additional reference.
111 */
112void force_yed_unload(void *handle) {
113#ifdef CAN_RELOAD_CORE
114 void *try_handle;
115
116 while ((try_handle = dlopen(lib_path, RTLD_LAZY | RTLD_NOLOAD))) {
117 dlclose(try_handle);
118 dlclose(handle);
119 }
120#else
121 dlclose(handle);
122#endif
123}
