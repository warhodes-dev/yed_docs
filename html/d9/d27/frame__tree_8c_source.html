177void yed_frame_tree_delete_leaf(yed_frame_tree *tree) {
178 yed_frame_tree *p;
179 yed_frame_tree *other_child;
180
181 if (!tree->is_leaf) { return; }
182
183 /*
184 * If it's just a root tree -- i.e. single frame with no children,
185 * then we just remove it.
186 */
187 if ((p = tree->parent) == NULL) { goto out_delete; }
188
189 other_child = p->child_trees[tree == p->child_trees[0]];
190
191 ASSERT(other_child, "leaf's parent doesn't have two children");
192
193 yed_frame_tree_set_relative_rect(other_child, p->top, p->left, p->height, p->width);
194
195 /* Replace p with other_child in p's parent tree. */
196 other_child->parent = p->parent;
197 if (p->parent != NULL) {
198 p->parent->child_trees[p != p->parent->child_trees[0]]
199 = other_child;
200 }
201
202 yed_delete_frame_tree(p);
203
204 yed_frame_tree_recursive_readjust(other_child);
205
206out_delete:
207 yed_delete_frame_tree(tree);
